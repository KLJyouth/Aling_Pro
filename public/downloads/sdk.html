快速开始

安装JavaScript SDK
npm install @alingai/sdk # 或者 yarn add @alingai/sdk
基础使用
import { AlingAI } from '@alingai/sdk'; const client = new AlingAI({ apiKey: 'your-api-key', quantumEnabled: true }); // 威胁检测 const result = await client.threatDetection.analyze({ data: logData, type: 'network_traffic' }); console.log(result);
安装Python SDK
pip install alingai-sdk
基础使用
from alingai import AlingAI client = AlingAI( api_key='your-api-key', quantum_enabled=True ) # 威胁检测 result = client.threat_detection.analyze( data=log_data, type='network_traffic' ) print(result)
Maven依赖
<dependency> <groupId>com.alingai</groupId> <artifactId>alingai-sdk</artifactId> <version>4.1.2</version> </dependency>
基础使用
import com.alingai.sdk.AlingAI; import com.alingai.sdk.config.Config; Config config = new Config.Builder() .apiKey("your-api-key") .quantumEnabled(true) .build(); AlingAI client = new AlingAI(config); // 威胁检测 ThreatResult result = client.threatDetection() .analyze(logData, "network_traffic"); System.out.println(result);
Go模块安装
go get github.com/alingai/go-sdk
基础使用
package main import ( "github.com/alingai/go-sdk" "fmt" ) func main() { client := alingai.NewClient(&alingai.Config{ APIKey: "your-api-key", QuantumEnabled: true, }) // 威胁检测 result, err := client.ThreatDetection.Analyze( logData, "network_traffic", ) if err != nil { fmt.Println("Error:", err) return } fmt.Println(result) }
