# AI威胁检测模型
class ThreatDetectionModel(nn.Module):
    def __init__(self, input_dim, hidden_dim, num_classes):
        super().__init__()
        self.cnn_layers = nn.Sequential(
            nn.Conv1d(input_dim, 64, kernel_size=3),
            nn.ReLU(),
            nn.Conv1d(64, 128, kernel_size=3),
            nn.ReLU(),
            nn.MaxPool1d(2)
        )
        
        self.lstm = nn.LSTM(128, hidden_dim, batch_first=True)
        self.classifier = nn.Linear(hidden_dim, num_classes)
        self.attention = MultiHeadAttention(hidden_dim, 8)
    
    def forward(self, x):
        # CNN特征提取
        x = self.cnn_layers(x)
        
        # LSTM序列建模
        lstm_out, _ = self.lstm(x.permute(0, 2, 1))
        
        # 注意力机制
        attended = self.attention(lstm_out)
        
        # 分类预测
        return self.classifier(attended[:, -1, :])
