JavaScript SDK

使用JavaScript调用AI API

JavaScript
// 安装 AlingAI SDK // npm install alingai-sdk import { AlingAI } from 'alingai-sdk'; // 初始化客户端 const client = new AlingAI({ apiKey: 'your-api-key', baseURL: 'https://api.alingai.com' }); // 调用AI对话API async function chatWithAI() { try { const response = await client.chat({ model: 'alingai-pro', messages: [ { role: 'user', content: '你好，AlingAI！' } ], temperature: 0.7, maxTokens: 1000 }); console.log(response.choices[0].message.content); } catch (error) { console.error('Error:', error); } } chatWithAI();
React 组件

在React应用中集成AI功能

React
import React, { useState } from 'react'; import { AlingAI } from 'alingai-sdk'; const ChatComponent = () => { const [message, setMessage] = useState(''); const [response, setResponse] = useState(''); const [loading, setLoading] = useState(false); const client = new AlingAI({ apiKey: process.env.REACT_APP_ALINGAI_API_KEY }); const handleSend = async () => { setLoading(true); try { const result = await client.chat({ model: 'alingai-pro', messages: [{ role: 'user', content: message }] }); setResponse(result.choices[0].message.content); } catch (error) { console.error(error); } setLoading(false); }; return ( <div className="chat-container"> <input value={message} onChange={(e) => setMessage(e.target.value)} placeholder="输入您的问题..." /> <button onClick={handleSend} disabled={loading}> {loading ? '发送中...' : '发送'} </button> {response && <div>{response}</div>} </div> ); }; export default ChatComponent;
Python SDK

使用Python调用AI API

Python
# 安装 AlingAI Python SDK # pip install alingai import alingai import os # 设置API密钥 alingai.api_key = os.getenv('ALINGAI_API_KEY') def chat_with_ai(prompt): """与AI进行对话""" try: response = alingai.ChatCompletion.create( model='alingai-pro', messages=[ {'role': 'system', 'content': '你是一个有帮助的AI助手。'}, {'role': 'user', 'content': prompt} ], temperature=0.7, max_tokens=1000 ) return response.choices[0].message.content except Exception as e: print(f'错误: {e}') return None # 示例使用 if __name__ == '__main__': user_input = input('请输入您的问题: ') ai_response = chat_with_ai(user_input) if ai_response: print(f'AI回答: {ai_response}') else: print('无法获取AI回应')
Flask Web应用

构建AI驱动的Web应用

Flask
from flask import Flask, request, jsonify, render_template import alingai import os app = Flask(__name__) alingai.api_key = os.getenv('ALINGAI_API_KEY') @app.route('/') def index(): return render_template('chat.html') @app.route('/api/chat', methods=['POST']) def chat_api(): try: data = request.get_json() user_message = data.get('message') if not user_message: return jsonify({'error': '消息不能为空'}), 400 response = alingai.ChatCompletion.create( model='alingai-pro', messages=[ {'role': 'user', 'content': user_message} ], temperature=0.7, max_tokens=1000 ) ai_message = response.choices[0].message.content return jsonify({ 'response': ai_message, 'status': 'success' }) except Exception as e: return jsonify({'error': str(e)}), 500 if __name__ == '__main__': app.run(debug=True, host='0.0.0.0', port=5000)
PHP SDK

在PHP应用中使用AI

PHP
// 安装 AlingAI PHP SDK // composer require alingai/sdk <?php require_once 'vendor/autoload.php'; use AlingAI\Client; class AIChatService { private $client; public function __construct($apiKey) { $this->client = new Client([ 'api_key' => $apiKey, 'base_url' => 'https://api.alingai.com' ]); } public function chat($message, $options = []) { try { $response = $this->client->chat([ 'model' => $options['model'] ?? 'alingai-pro', 'messages' => [ ['role' => 'user', 'content' => $message] ], 'temperature' => $options['temperature'] ?? 0.7, 'max_tokens' => $options['max_tokens'] ?? 1000 ]); return [ 'success' => true, 'data' => $response['choices'][0]['message']['content'] ]; } catch (Exception $e) { return [ 'success' => false, 'error' => $e->getMessage() ]; } } } // 使用示例 $apiKey = getenv('ALINGAI_API_KEY'); $aiService = new AIChatService($apiKey); $userMessage = "你好，请介绍一下自己"; $result = $aiService->chat($userMessage); if ($result['success']) { echo "AI回答: " . $result['data']; } else { echo "错误: " . $result['error']; } ?>
Laravel集成

在Laravel框架中使用AI

Laravel
// app/Http/Controllers/ChatController.php <?php namespace App\Http\Controllers; use Illuminate\Http\Request; use AlingAI\Client; class ChatController extends Controller { protected $aiClient; public function __construct() { $this->aiClient = new Client([ 'api_key' => config('services.alingai.api_key'), 'base_url' => config('services.alingai.base_url') ]); } public function chat(Request $request) { $request->validate([ 'message' => 'required|string|max:1000' ]); try { $response = $this->aiClient->chat([ 'model' => 'alingai-pro', 'messages' => [ ['role' => 'user', 'content' => $request->message] ], 'temperature' => 0.7 ]); return response()->json([ 'success' => true, 'data' => [ 'message' => $response['choices'][0]['message']['content'], 'timestamp' => now() ] ]); } catch (\Exception $e) { return response()->json([ 'success' => false, 'error' => 'AI服务暂时不可用' ], 500); } } public function index() { return view('chat.index'); } } // config/services.php 添加配置 'alingai' => [ 'api_key' => env('ALINGAI_API_KEY'), 'base_url' => env('ALINGAI_BASE_URL', 'https://api.alingai.com') ],
cURL 示例

直接使用HTTP请求调用API

cURL
# 基本的聊天API调用 curl -X POST https://api.alingai.com/v1/chat/completions \ -H "Content-Type: application/json" \ -H "Authorization: Bearer YOUR_API_KEY" \ -d '{ "model": "alingai-pro", "messages": [ { "role": "user", "content": "你好，请介绍一下AlingAI" } ], "temperature": 0.7, "max_tokens": 1000 }' # 流式响应调用 curl -X POST https://api.alingai.com/v1/chat/completions \ -H "Content-Type: application/json" \ -H "Authorization: Bearer YOUR_API_KEY" \ -d '{ "model": "alingai-pro", "messages": [ { "role": "user", "content": "写一首关于人工智能的诗" } ], "temperature": 0.8, "stream": true }' # 文本生成API调用 curl -X POST https://api.alingai.com/v1/completions \ -H "Content-Type: application/json" \ -H "Authorization: Bearer YOUR_API_KEY" \ -d '{ "model": "alingai-text", "prompt": "人工智能的未来发展趋势", "max_tokens": 500, "temperature": 0.6 }' # 图像生成API调用 curl -X POST https://api.alingai.com/v1/images/generations \ -H "Content-Type: application/json" \ -H "Authorization: Bearer YOUR_API_KEY" \ -d '{ "prompt": "一个美丽的未来科技城市，赛博朋克风格", "n": 1, "size": "1024x1024" }'
