5 分钟快速集成

JavaScript 

// 安装 SDK
npm install @alingai/sdk

// 初始化客户端
import { AlingAI } from '@alingai/sdk';

const client = new AlingAI({
  apiKey: 'your-api-key'
});

// 发送对话请求
const response = await client.chat.create({
  model: 'alingai-gpt-4',
  messages: [
    { role: 'user', content: 'Hello, AlingAI!' }
  ]
});

console.log(response.choices[0].message.content);

Python 

# 安装 SDK
pip install alingai

# 导入和初始化
from alingai import AlingAI

client = AlingAI(api_key="your-api-key")

# 发送对话请求
response = client.chat.create(
    model="alingai-gpt-4",
    messages=[
        {"role": "user", "content": "Hello, AlingAI!"}
    ]
)

print(response.choices[0].message.content)
