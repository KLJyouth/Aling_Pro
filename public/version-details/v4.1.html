智能分析API示例

Python SDK - 威胁预测 
from alingai.analysis import ThreatPredictor
from alingai.models import SecurityEvent

# 初始化威胁预测器
predictor = ThreatPredictor(
    model='neural_network_v2',
    confidence_threshold=0.85
)

# 加载历史安全事件数据
events = SecurityEvent.load_from_database(
    start_date='2024-01-01',
    end_date='2024-04-01'
)

# 训练预测模型
predictor.train(events)

# 预测未来7天的威胁
predictions = predictor.predict_threats(
    horizon_days=7,
    include_confidence=True
)

for prediction in predictions:
    print(f"日期: {prediction.date}")
    print(f"威胁类型: {prediction.threat_type}")
    print(f"风险等级: {prediction.risk_level}")
    print(f"置信度: {prediction.confidence:.2%}")
    print("---")
