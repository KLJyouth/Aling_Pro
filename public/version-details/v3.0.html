企业级部署命令 

# 1. 初始化多租户环境
./alingai-enterprise-init.sh --multi-tenant --enterprise-mode

# 2. 创建企业租户
./alingai-tenant.sh --create --name="企业A" --type=enterprise

# 3. 配置企业安全策略
./alingai-security.sh --apply-policy --file=enterprise-security.yml

# 4. 启用高级审计
./alingai-audit.sh --enable --level=enterprise --retention=365d

# 5. 配置企业集成
./alingai-integration.sh --enable-erp --enable-crm --enable-hr
