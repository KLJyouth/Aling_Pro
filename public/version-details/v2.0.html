v2.0 迁移命令 

# 1. 备份v1.x数据
./backup-v1.sh --full-backup --export-config

# 2. 部署v2.0环境
docker-compose -f docker-compose.v2.yml up -d

# 3. 执行数据迁移
./migrate-to-v2.sh --from-v1 --validate-data

# 4. 导入配置
./import-config.sh --from-v1-backup

# 5. 启动所有微服务
kubectl apply -f k8s/v2.0-manifests/

# 6. 验证迁移结果
./verify-migration.sh --comprehensive-test
