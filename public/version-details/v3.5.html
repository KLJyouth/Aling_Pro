升级命令 

# 1. 备份当前版本
sudo ./alingai-backup.sh --full

# 2. 下载升级包
wget https://releases.alingai.com/v3.5/alingai-v3.5-enterprise.tar.gz

# 3. 执行升级
sudo ./alingai-upgrade.sh --from=v3.0 --to=v3.5 --preserve-config

# 4. 验证升级
./alingai-verify.sh --version=v3.5
